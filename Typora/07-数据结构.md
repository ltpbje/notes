		LNode* pa = La->next;
		LNode* pb = Lb->next;
		Lc = La;
		LNode* pc = Lc;
		while (pa&&pb) {
			if (pa->data < pb->data) {
				pc->next = pa;
				pc = pa;
				pa = pa->next;
